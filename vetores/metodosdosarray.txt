        //criando uma array de tipo inteiro e declarando valores
        int[] num = {10, 20, 30, 40, 50};

        //demonstração de um for tradicional
        for(int i=0; i<num.length; i++){

            System.out.printf("%d - ", num[i]);
        }

        //for especifico para ler os elementos de uma array
        for(int n:num){
            
            System.out.printf("%d - ", n);
        }

        Arrays.sort(num); // ordena/organiza os numeros de uma array(crescente)
